244: # 2. Khai báo hàm async tối cao để Orchestrator thực hiện await
245: async def search_market(topic: str) -> dict:
246:     year = extract_year(topic) 
247:     
248:     # Sử dụng context manager httpx.AsyncClient để quản lý connection pool phi nghẽn mạng
249:     async with httpx.AsyncClient() as client:
250:         # Tạo cấu trúc luồng chạy song song (Parallel tasks) cho cả 3 từ khóa
251:         task_overview = search(f"{topic} tổng quan thị trường", client)
252:         task_trends = search(f"{topic} xu hướng {year}", client)
253:         task_competitors = search(f"{topic} doanh nghiệp thương hiệu lớn {year}", client)
254:         
255:         # Đồng loạt phát lệnh kích hoạt mạng phi trạng thái và thu hồi kết quả cùng một lúc
256:         overview, trends, competitors = await asyncio.gather(
257:             task_overview, 
258:             task_trends, 
259:             task_competitors
260:         )
261:     
262:     # Tổ chức lại cấu trúc dữ liệu trả về tương thích 100% với trạng thái memory cũ
263:     results = {}
264:     results["overview"] = overview["text"]
265:     results["trends"] = trends["text"]
266:     results["competitors"] = competitors["text"]
267:     results["overview_sources"] = overview["sources"]
268:     results["trends_sources"] = trends["sources"]
269:     results["competitors_sources"] = competitors["sources"]
270:     results["sources"] = (
271:         overview["sources"] +
272:         trends["sources"] +
273:         competitors["sources"]
274:     )
275:     results["html_pages"] = [
276:         {"label": "Overview", "url": overview.get("html_url", "")},
277:         {"label": "Trends", "url": trends.get("html_url", "")},
278:         {"label": "Competitors", "url": competitors.get("html_url", "")}
279:     ]
280:     return results
